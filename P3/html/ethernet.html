116def registerCallback(callback_func: Callable[[ctypes.c_void_p,pcap_pkthdr,bytes],None], ethertype:int) -> None:
117    '''
118        Nombre: registerCallback
119        Descripción: Esta función recibirá el nombre de una función y su valor de ethertype asociado y añadirá en la tabla 
120            (diccionario) de protocolos de nivel superior el dicha asociación. 
121            Este mecanismo nos permite saber a qué función de nivel superior debemos llamar al recibir una trama de determinado tipo. 
122            Por ejemplo, podemos registrar una función llamada process_IP_datagram asociada al Ethertype 0x0800 y otra llamada process_arp_packet 
123            asocaida al Ethertype 0x0806. 
124        Argumentos:
125            -callback_fun: función de callback a ejecutar cuando se reciba el Ethertype especificado. 
126                La función que se pase como argumento debe tener el siguiente prototipo: funcion(us,header,data,srcMac)
127                Dónde:
128                    -us: son los datos de usuarios pasados por pcap_loop (en nuestro caso este valor será siempre None)
129                    -header: estructura pcap_pkthdr que contiene los campos len, caplen y ts.
130                    -data: payload de la trama Ethernet. Es decir, la cabecera Ethernet NUNCA se pasa hacia arriba.
131                    -srcMac: dirección MAC que ha enviado la trama actual.
132                La función no retornará nada. Si una trama se quiere descartar basta con hacer un return sin valor y dejará de procesarse.
133            -ethertype: valor de Ethernetype para el cuál se quiere registrar una función de callback.
134        Retorno: Ninguno 
135    '''
136    global upperProtos
137    #upperProtos es el diccionario que relaciona función de callback y ethertype
138    upperProtos[str(ethertype)] = callback_func

140def startEthernetLevel(interface:str) -> int:
141    '''
142        Nombre: startEthernetLevel
143        Descripción: Esta función recibe el nombre de una interfaz de red e inicializa el nivel Ethernet. 
144            Esta función debe realizar , al menos, las siguientes tareas:
145                -Comprobar si el nivel Ethernet ya estaba inicializado (mediante una variable global). Si ya estaba inicializado devolver -1.
146                -Obtener y almacenar en una variable global la dirección MAC asociada a la interfaz que se especifica
147                -Abrir la interfaz especificada en modo promiscuo usando la librería rc1-pcap
148                -Arrancar un hilo de recepción (rxThread) que llame a la función pcap_loop. 
149                -Si todo es correcto marcar la variable global de nivel incializado a True
150        Argumentos:
151            -Interface: nombre de la interfaz sobre la que inicializar el nivel Ethernet
152        Retorno: 0 si todo es correcto, -1 en otro caso
153    '''
154    global macAddress,handle,levelInitialized,recvThread
155    handle = None
156    errbuff = bytearray() # Se crea una variable errbuff para pcap_open_live.
157
158    try:
159        if levelInitialized == True:
160            return -1
161    except:
162        levelInitialized = False
163
164    macAddress = getHwAddr(interface)
165    handle = pcap_open_live(interface, ETH_FRAME_MAX, PROMISC, TO_MS, errbuff)
166    if handle is None:
167        return -1
168
169    levelInitialized = True # Se ha inicializado de forma correcta el nivel de interfaz.
170    
171    #Una vez hemos abierto la interfaz para captura y hemos inicializado las variables globales (macAddress, handle y levelInitialized) arrancamos
172    #el hilo de recepción
173    recvThread = rxThread()
174    recvThread.daemon = True
175    recvThread.start()
176    return 0

201def sendEthernetFrame(data:bytes,length:int,etherType:int,dstMac:bytes) -> int:
202    '''
203        Nombre: sendEthernetFrame
204        Descripción: Esta función construirá una trama Ethernet con lo datos recibidos y la enviará por la interfaz de red. 
205            Esta función debe realizar, al menos, las siguientes tareas:
206                -Construir la trama Ethernet a enviar (incluyendo cabecera + payload). Los campos propios (por ejemplo la dirección Ethernet origen) 
207                    deben obtenerse de las variables que han sido inicializadas en startEthernetLevel
208                -Comprobar los límites de Ethernet. Si la trama es muy pequeña se debe rellenar con 0s mientras que 
209                    si es muy grande se debe devolver error.
210                -Llamar a pcap_inject para enviar la trama y comprobar el retorno de dicha llamada. En caso de que haya error notificarlo
211        Argumentos:
212            -data: datos útiles o payload a encapsular dentro de la trama Ethernet
213            -length: longitud de los datos útiles expresada en bytes
214            -etherType: valor de tipo Ethernet a incluir en la trama
215            -dstMac: Dirección MAC destino a incluir en la trama que se enviará
216        Retorno: 0 si todo es correcto, -1 en otro caso
217    '''
218    global macAddress,handle
219
220    # Construimos la trama ethernet en bytes: dirección destino + dirección actual + tipo de ethernet + payload.
221    ethFrame = dstMac + macAddress + struct.pack('!H', etherType) + data # !H introduce el entero como bytes en BIGENDIAN.
222    tam = 14 + length # La cabecera Ethernet ocupa 14 bytes. La trama es cabecera + payload.
223
224    # Si la trama es demasiado grande devolvemos error.
225    if tam > ETH_FRAME_MAX:
226        logging.debug('Trama más grande de lo permitida.')
227        return -1
228
229    # Si la trama es demasiado corta la rellenamos con bytes de '0' y actualizamos el tam.
230    if tam < ETH_FRAME_MIN:
231        ethFrame = ethFrame.ljust(ETH_FRAME_MIN, b'\0')
232        tam = ETH_FRAME_MIN
233
234    try:
235        pcap_inject(handle, ethFrame, tam)
236    except ValueError:
237        logging.debug('Error en los argumentos del pcap_inject().')
238        return -1
239
240    return 0
